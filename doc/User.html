def self.find_by_credentials(username, password)
  user = User.find_by(username: username)
  return nil unless user&.valid_password?(password)
  user
end

def average_listing_rating
  avg = Listing.
    joins(:reviews).
    where(lessor_id: id).
    average("reviews.review")

  avg&.round(0).to_i
end
